function _solidityBaselineRepDelta(uint8 conviction, uint8 /*consensusPct*/, bool isWinner)
    internal pure returns (int256 delta) {
    uint256 BASE = 10;
    uint256 scaled = (BASE * uint256(conviction) * 2) / 100;

    delta = isWinner ? int256(scaled) : -int256(scaled);
